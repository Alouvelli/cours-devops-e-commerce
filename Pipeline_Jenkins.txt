pipeline {
    agent any
    tools {
    maven 'MAVEN_HOME'
    }
    stages {
        stage('Get source') {
            steps {
                git(
                    url:'https://github.com/bassbiteye/cours-devops-e-commerce',
                    credentialsId:'github_id_isi',
                    branch:'utilisateur'
                )
            }
        }

        stage('JUnit test'){
             steps{
                bat "mvn clean test"
             }
        }
        stage('Clean and Install'){
            steps{
                bat "mvn clean install -f pom.xml"
            }
        }

        stage ('Scan Code Quality')  {
             steps{
               withSonarQubeEnv('SonarQube') {
               bat "mvn -f pom.xml sonar:sonar"
                }
             }
        }



          stage ('Local Deployment ')  {
            steps{
            rtServer (
                id:"Artifactory",
                url:"http://localhost:8082/artifactory",
                username:"admin",
                password:"password",
                bypassProxy:true,
                timeout:300

                )
        }
        }
       stage('Upload') {
        steps{
            rtUpload (
                serverId:"Artifactory",
               spec: '''{
                   "files":[
                       {
                           "pattern":"*.war",
                           "target": "ecomerce-devops-libs-snapshot-local"
                       }
                       ]
               }''',

            )
        }
        }
        stage('Publish build info') {
        steps{
            rtPublishBuildInfo (
                serverId:"Artifactory"

                )
        }
        }
    }
}
