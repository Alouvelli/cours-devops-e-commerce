    agent any
    environment {
        PATH = "$PATH:C:\\soft\\apache-maven-3.8.4\\bin"
    }
    stages{
       stage('Clone'){
            steps{
               git(url:'https://github.com/bassbiteye/cours-devops-e-commerce',

            steps{

               }
           }

    }
        }
        }
       
    }
